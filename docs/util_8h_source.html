95 static int cecies_hexstr2bin(const char* hexstr, const size_t hexstr_length, unsigned char* output, const size_t output_size, size_t* output_length)
96 {
97  if (hexstr == NULL || output == NULL || hexstr_length == 0)
98  {
99  return 1;
100  }
101 
102  const size_t hl = hexstr[hexstr_length - 1] ? hexstr_length : hexstr_length - 1;
103 
104  if (hl % 2 != 0)
105  {
106  return 2;
107  }
108 
109  const size_t final_length = hl / 2;
110 
111  if (output_size < final_length + 1)
112  {
113  return 3;
114  }
115 
116  for (size_t i = 0, ii = 0; ii < final_length; i += 2, ii++)
117  {
118  output[ii] = (hexstr[i] % 32 + 9) % 25 * 16 + (hexstr[i + 1] % 32 + 9) % 25;
119  }
120 
121  output[final_length] = '\0';
122 
123  if (output_length != NULL)
124  {
125  *output_length = final_length;
126  }
127 
128  return 0;
129 }

143 {
144  if (bin == NULL || bin_length == 0 || output == NULL)
145  {
146  return 1;

148 
149  const size_t final_length = bin_length * 2;
150 
151  if (output_size < final_length + 1)

153  return 2;

157
